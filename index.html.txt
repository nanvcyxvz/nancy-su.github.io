    <script>
        const { createApp, ref } = Vue;
        createApp({
            setup() {
                const itinerary = ref([
                    { time: '10:00', location: 'Tokyo Tower' },
                    { time: '14:30', location: 'Shibuya Sky' }
                ]);
                const bill = ref({ jpy: 12000, people: 2 });
                const rate = ref(0.21);
                const weather = ref({ temp: 12, condition: 'Clear' });

                const addItem = () => itinerary.value.push({ time: '12:00', location: '' });
                const removeItem = (idx) => itinerary.value.splice(idx, 1);
                
                const openGoogleMaps = (loc) => {
                    if(!loc) return;
                    window.open(`https://www.google.com/maps/search/?api=1&query=${encodeURIComponent(loc)}`, '_blank');
                };

                const openAllLocations = () => {
                    const locs = itinerary.value.filter(i => i.location).map(i => i.location).join('/');
                    window.open(`https://www.google.com/maps/dir/${locs}`, '_blank');
                };

                const exportPDF = () => {
                    const element = document.getElementById('app-content');
                    html2pdf().from(element).set({
                        margin: 10,
                        filename: 'My_Quiet_Journey.pdf',
                        html2canvas: { scale: 2 },
                        jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
                    }).save();
                };

                return { itinerary, bill, rate, weather, addItem, removeItem, openGoogleMaps, openAllLocations, exportPDF };
            }
        }).mount('#app');
    </script>
